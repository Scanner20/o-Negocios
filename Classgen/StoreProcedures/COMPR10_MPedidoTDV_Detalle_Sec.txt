CREATE PROCEDURE COMPR10_MPedidoTDV_Detalle_Sec
	@NumeroPedidoTDV		   char(8) = NULL , 	@NumeroPedidoTDV_Sub		   char(3) = NULL , 	@Item				  char(3) = NULL , 	@CodigoSector			  char(6) = NULL , 	@FHInicioProgramado		datetime = NULL , 	@FHTerminoProgramado		   datetime = NULL , 	@FHInicioReal			  datetime = NULL , 	@FHTerminoReal			 datetime = NULL , 
	@Usuario		varchar(15) = NULL ,
	@Estacion		varchar(15) = NULL ,
	@TipoOperacion	varchar(2)	= 'C1'
AS
BEGIN  -- Inicion de Bloque Principal

	
	DECLARE @Mensaje		varchar(200)

	SET @Mensaje = 'Sin Errores...'
	

	IF @Usuario	= NULL		SET @Usuario	= SUSER_NAME()
	IF @Estacion	= NULL		SET @Estacion	= HOST_NAME()
	

	BEGIN TRANSACTION Trans_MPedidoTDV_Detalle_Secto

	-- Insertar Elementos 
	IF @TipoOperacion='I'
	BEGIN
		/*
		-- Obtener un correlativo valido para la SolicitudDDP
		EXEC ADMGEN4_Contador_Entidad @Entidad = 'MPedidoTDV_Detalle_Sectores', @Contador = @Variable OUTPUT, @Actualizar = 1

		IF @@ERROR <> 0 	-- Ocurrio un error al obtener el correlativo
		BEGIN
			SET @Mensaje = 'Fallo el correlativo para la Tabla :  MPedidoTDV_Detalle_Sectores' 
			GOTO ERROR	-- Hubo un error en la Inserción
		END 

		-- Obtener un Correlativo Valido para el Item
		SELECT @Variable = COALESCE( CAST( MAX( <Campo> ) + 1 AS CHAR(3) ) , '001' ) 
		FROM MPedidoTDV_Detalle_Sectores WHERE NumeroPedidoTDV = @NumeroPedidoTDV AND 			NumeroPedidoTDV_Sub = @NumeroPedidoTDV_Sub AND 			Item = @Item 

		SET @NumeroSCot_Orden = RIGHT( '000' + RTRIM( @NumeroSCot_Orden ) , 3 )

		*/

		INSERT INTO MPedidoTDV_Detalle_Sectores
			( NumeroPedidoTDV , 			NumeroPedidoTDV_Sub , 			Item , 			CodigoSector , 			FHInicioProgramado , 			FHTerminoProgramado , 			FHInicioReal , 			FHTerminoReal , 			
			FHCreacion , UsuarioCreacion , EstacionCreacion , 
			FHModificacion , UsuarioModificacion , EstacionModificacion )
		VALUES 
			( @NumeroPedidoTDV , 			@NumeroPedidoTDV_Sub , 			@Item , 			@CodigoSector , 			@FHInicioProgramado , 			@FHTerminoProgramado , 			@FHInicioReal , 			@FHTerminoReal , 			
			GETDATE() , @Usuario , @Estacion ,
			GETDATE() , @Usuario , @Estacion )

		IF @@ERROR <> 0 	-- Ocurrio un error
		BEGIN
			SET @Mensaje = 'Ocurrió un error al Insertar un elemento en la Tabla : MPedidoTDV_Detalle_Sectores'
			GOTO ERROR
		END

		GOTO OK
	END

	-- Actualizar Elementos 
	IF @TipoOperacion='A'
	BEGIN
		UPDATE MPedidoTDV_Detalle_Sectores SET 
			CodigoSector = @CodigoSector , 			FHInicioProgramado = @FHInicioProgramado , 			FHTerminoProgramado = @FHTerminoProgramado , 			FHInicioReal = @FHInicioReal , 			FHTerminoReal = @FHTerminoReal , 
			FHModificacion 		= GETDATE() , 
			UsuarioModificacion	= @Usuario , 
			EstacionModificacion= @Estacion
		WHERE 
			NumeroPedidoTDV = @NumeroPedidoTDV AND 			NumeroPedidoTDV_Sub = @NumeroPedidoTDV_Sub AND 			Item = @Item  AND
			FlagEliminado = 0

		IF @@ERROR <> 0 	-- Ocurrio un error
		BEGIN
			SET @Mensaje = 'Ocurrió un error al Actualizar la Tabla : MPedidoTDV_Detalle_Sectores'
			GOTO ERROR
		END

		GOTO OK
	END

	-- Eliminar Elementos 
	IF @TipoOperacion='E'
	BEGIN
		UPDATE MPedidoTDV_Detalle_Sectores SET 
			FlagEliminado 		= 1 ,
			FHModificacion 		= GETDATE() , 
			UsuarioModificacion	= @Usuario , 
			EstacionModificacion= @Estacion
		WHERE 
			NumeroPedidoTDV = @NumeroPedidoTDV AND 			NumeroPedidoTDV_Sub = @NumeroPedidoTDV_Sub AND 			Item = @Item 

		IF @@ERROR <> 0 	-- Ocurrio un error
		BEGIN
			SET @Mensaje = 'Ocurrió un error al Eliminar un elemento en la Tabla : MPedidoTDV_Detalle_Sectores'
			GOTO ERROR
		END

		GOTO OK
	END

	-- Consultar Elementos 
	IF @TipoOperacion='C1'
	BEGIN
		SELECT 
			NumeroPedidoTDV , 			NumeroPedidoTDV_Sub , 			Item , 			CodigoSector , 			FHInicioProgramado , 			FHTerminoProgramado , 			FHInicioReal , 			FHTerminoReal 			
		FROM 
			MPedidoTDV_Detalle_Sectores
		WHERE 
			NumeroPedidoTDV = @NumeroPedidoTDV AND 			NumeroPedidoTDV_Sub = @NumeroPedidoTDV_Sub AND 			Item = @Item  AND
			FlagEliminado = 0

		IF @@ERROR <> 0 	-- Ocurrio un error
		BEGIN
			SET @Mensaje = 'Ocurrió un error al consultar la Tabla : MPedidoTDV_Detalle_Sectores'
			GOTO ERROR
		END

		GOTO OK
	END


ERROR:
	ROLLBACK TRANSACTION Trans_MPedidoTDV_Detalle_Secto
	RAISERROR(@Mensaje,18,1)
	GOTO FIN

OK:

	COMMIT TRANSACTION Trans_MPedidoTDV_Detalle_Secto

FIN:
END -- Fin de Bloque Principal
